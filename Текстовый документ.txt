print('Введите первое число:')
x = int(input())
print('Введите второе число:')
y = int(input())

print('Введите арифметический знак (+, -, *, /):')
str = input()

#Далее условие для распознания арифметичского знака (+ - * /)

if str == '+':
    print('Результат сложения:', x + y)
elif str == '-':
    print('Результат вычетания:', x - y)
elif str == '*':
    print(x * y)
elif str == '/' and y != 0:
    print(x / y)
else:
    print('Ошибка!')
